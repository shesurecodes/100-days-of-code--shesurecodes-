# Rock Paper Scissors
import random
from random import randint

rock = '''
    _______
---'   ____)
      (_____)
      (_____)
      (____)
---.__(___)
'''

paper = '''
    _______
---'   ____)____
          ______)
          _______)
         _______)
---.__________)
'''

scissors = '''
    _______
---'   ____)____
          ______)
       __________)
      (____)
---.__(___)
'''
rock_paper_scissor = int(input("What do you choose? Type 0 for Rock, 1 for Paper and 2 for Scissors."))
if rock_paper_scissor == 0:
    print(rock)
if rock_paper_scissor == 1:
    print(paper)
if rock_paper_scissor == 2:
    print(scissors)
print("Computer chose:")
random_rock_paper_scissors = random.randint(0, 2)
if rock_paper_scissor == 0 and random_rock_paper_scissors == 0:
    print(rock)
    print("It\'s a draw.")
if rock_paper_scissor == 0 and random_rock_paper_scissors == 1:
    print(paper)
    print("You lose.")
if rock_paper_scissor == 0 and random_rock_paper_scissors == 2:
    print(scissors)
    print("You win!")
if rock_paper_scissor == 1 and random_rock_paper_scissors == 1:
    print(paper)
    print("It\'s a draw.")
if rock_paper_scissor == 1 and random_rock_paper_scissors == 0:
    print(rock)
    print("You win!")
if rock_paper_scissor == 1 and random_rock_paper_scissors == 2:
    print(scissors)
    print("You lose!")
if rock_paper_scissor == 2 and random_rock_paper_scissors == 2:
    print(scissors)
    print("It\'s a draw.")
if rock_paper_scissor == 2 and random_rock_paper_scissors == 0:
    print(rock)
    print("You lose!")
if rock_paper_scissor == 2 and random_rock_paper_scissors == 1:
    print(paper)
    print("You lose!")
